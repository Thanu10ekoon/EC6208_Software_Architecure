# SmartFines Database Architecture (DBML)

```dbml
Enum user_status {
  active
  suspended
  deleted
}

Enum license_status {
  active
  suspended
  cancelled
}

Enum fine_status {
  issued
  paid
  disputed
  cancelled
  void
}

Enum payment_status {
  pending
  paid
  failed
  reversed
}

Enum payment_method {
  online
  receipt_upload
}

Enum receipt_source {
  web_generated
  driver_uploaded
}

Enum notification_channel {
  in_app
  email
  sms
}

Enum notification_delivery_status {
  queued
  sent
  failed
}

Enum recollection_status {
  pending
  marked_by_driver
  confirmed
}

Table users {
  id bigint [pk, increment]
  full_name varchar(150) [not null]
  username varchar(50) [unique, note: "Admin login username"]
  email varchar(255) [unique, not null]
  phone varchar(30) [unique]
  nic varchar(30) [unique, note: "Driver login identifier"]
  password_hash varchar(255) [not null]
  status user_status [not null, default: 'active']
  last_login_at timestamp
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
  deleted_at timestamp
}

Table roles {
  id bigint [pk, increment]
  name varchar(50) [not null, unique, note: "Examples: admin, traffic_officer, driver"]
  description varchar(255)
  created_at timestamp [not null, default: `now()`]
}

Table permissions {
  id bigint [pk, increment]
  code varchar(100) [not null, unique, note: "Examples: officer.create_fine, admin.add_officer"]
  description varchar(255)
  created_at timestamp [not null, default: `now()`]
}

Table user_roles {
  user_id bigint [not null]
  role_id bigint [not null]
  assigned_by_user_id bigint
  assigned_at timestamp [not null, default: `now()`]

  indexes {
    (user_id, role_id) [pk]
  }
}

Table role_permissions {
  role_id bigint [not null]
  permission_id bigint [not null]
  granted_at timestamp [not null, default: `now()`]

  indexes {
    (role_id, permission_id) [pk]
  }
}

Table regions {
  id bigint [pk, increment]
  code varchar(30) [not null, unique]
  name varchar(100) [not null]
  district varchar(100)
  province varchar(100)
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table license_details {
  user_id bigint [pk, note: "1:1 with users"]
  license_number varchar(50) [not null, unique]
  date_of_birth date [not null]
  address text
  region_id bigint
  stars int [not null, default: 5, note: "Starts at 5, decrements per fine"]
  license_status license_status [not null, default: 'active']
  license_cancelled_at timestamp
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table vehicle_categories {
  id bigint [pk, increment]
  code varchar(20) [not null, unique, note: "Example: A, B, C, D, G1"]
  name varchar(100) [not null]
  description varchar(255)
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table license_detail_categories {
  license_detail_user_id bigint [not null]
  vehicle_category_id bigint [not null]
  granted_at timestamp [not null, default: `now()`]
  expires_at timestamp

  indexes {
    (license_detail_user_id, vehicle_category_id) [pk]
  }
}

Table officer_profiles {
  user_id bigint [pk, note: "1:1 with users"]
  officer_code varchar(50) [not null, unique]
  badge_number varchar(50) [unique]
  station_name varchar(150)
  region_id bigint
  created_by_user_id bigint [note: "Admin who added this officer"]
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table traffic_fines {
  id bigint [pk, increment]
  fine_reference_number varchar(80) [not null, unique, note: "Auto-generated by the system"]
  driver_user_id bigint [not null]
  officer_user_id bigint [not null]
  region_id bigint [not null, note: "Region where violation happened"]
  vehicle_number varchar(30) [not null]
  driver_license_number_snapshot varchar(50) [not null, note: "Stored as a snapshot at the time of issuance"]
  violation_date date [not null]
  violation_details text [not null]
  violation_place varchar(255) [not null]
  fine_amount decimal(10,2) [not null]
  license_collection_location varchar(255) [not null]
  status fine_status [not null, default: 'issued']
  issued_at timestamp [not null, default: `now()`]
  paid_at timestamp
  due_at timestamp
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table fine_status_history {
  id bigint [pk, increment]
  fine_id bigint [not null]
  previous_status fine_status
  new_status fine_status [not null]
  changed_by_user_id bigint [not null]
  changed_at timestamp [not null, default: `now()`]
  comment varchar(255)
}

Table driver_star_history {
  id bigint [pk, increment]
  driver_user_id bigint [not null]
  fine_id bigint
  stars_before int [not null]
  stars_after int [not null]
  change_reason varchar(255) [not null, note: "Example: fine issued"]
  changed_by_user_id bigint [not null]
  changed_at timestamp [not null, default: `now()`]
}

Table payments {
  id bigint [pk, increment]
  fine_id bigint [not null]
  driver_user_id bigint [not null]
  amount decimal(10,2) [not null]
  payment_method payment_method [not null]
  payment_status payment_status [not null, default: 'pending']
  transaction_reference varchar(100) [unique]
  paid_at timestamp
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table payment_receipts {
  id bigint [pk, increment]
  payment_id bigint [not null, unique]
  receipt_number varchar(80) [not null, unique]
  source receipt_source [not null]
  file_url text [not null]
  file_name varchar(255)
  mime_type varchar(100)
  file_size_bytes bigint
  uploaded_by_user_id bigint
  generated_at timestamp
  verified_by_user_id bigint
  verified_at timestamp
  notes varchar(255)
  created_at timestamp [not null, default: `now()`]
}

Table license_recollections {
  id bigint [pk, increment]
  fine_id bigint [not null, unique]
  driver_user_id bigint [not null]
  status recollection_status [not null, default: 'pending']
  marked_recollected_at timestamp
  confirmed_by_user_id bigint
  confirmed_at timestamp
  notes varchar(255)
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp
}

Table notifications {
  id bigint [pk, increment]
  recipient_user_id bigint [not null]
  actor_user_id bigint
  type varchar(80) [not null, note: "Example: payment_received, receipt_uploaded, license_recollected"]
  title varchar(150) [not null]
  message text [not null]
  related_fine_id bigint
  related_payment_id bigint
  action_url text
  is_read boolean [not null, default: false]
  read_at timestamp
  created_at timestamp [not null, default: `now()`]
}

Table notification_deliveries {
  id bigint [pk, increment]
  notification_id bigint [not null]
  channel notification_channel [not null]
  delivery_status notification_delivery_status [not null, default: 'queued']
  delivered_to varchar(255)
  sent_at timestamp
  error_message text
  created_at timestamp [not null, default: `now()`]
}

Ref: user_roles.user_id > users.id
Ref: user_roles.role_id > roles.id
Ref: user_roles.assigned_by_user_id > users.id

Ref: role_permissions.role_id > roles.id
Ref: role_permissions.permission_id > permissions.id

Ref: license_details.user_id > users.id
Ref: license_details.region_id > regions.id

Ref: license_detail_categories.license_detail_user_id > license_details.user_id
Ref: license_detail_categories.vehicle_category_id > vehicle_categories.id

Ref: officer_profiles.user_id > users.id
Ref: officer_profiles.region_id > regions.id
Ref: officer_profiles.created_by_user_id > users.id

Ref: traffic_fines.driver_user_id > users.id
Ref: traffic_fines.officer_user_id > users.id
Ref: traffic_fines.region_id > regions.id

Ref: fine_status_history.fine_id > traffic_fines.id
Ref: fine_status_history.changed_by_user_id > users.id

Ref: driver_star_history.driver_user_id > users.id
Ref: driver_star_history.fine_id > traffic_fines.id
Ref: driver_star_history.changed_by_user_id > users.id

Ref: payments.fine_id > traffic_fines.id
Ref: payments.driver_user_id > users.id

Ref: payment_receipts.payment_id > payments.id
Ref: payment_receipts.uploaded_by_user_id > users.id
Ref: payment_receipts.verified_by_user_id > users.id

Ref: license_recollections.fine_id > traffic_fines.id
Ref: license_recollections.driver_user_id > users.id
Ref: license_recollections.confirmed_by_user_id > users.id

Ref: notifications.recipient_user_id > users.id
Ref: notifications.actor_user_id > users.id
Ref: notifications.related_fine_id > traffic_fines.id
Ref: notifications.related_payment_id > payments.id

Ref: notification_deliveries.notification_id > notifications.id
```
